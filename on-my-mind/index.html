Bytes are weird... but the Rust programming language is cool.

fn main() -> Result<()> {
    let f = File::create("bytes.txt")?;
    {
        let bytes: Vec<u8> = (0..=255).collect();
        let mut writer = BufWriter::new(f);
        match writer.write_all(&bytes) {
            Ok(()) => (),
            Err(e) => println!("ERR: {}", e),
        }
    }
    Ok(())
}
